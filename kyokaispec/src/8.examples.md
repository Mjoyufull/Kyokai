# Example Programs

Each runnable program pairs a **module interface** (exporting `main`) with a
**module body**, as required by the compiler for the program entrypoint.

## Small Program

```austral
module Example is
    function main(): ExitCode;
end module.

module body Example is
    function main(): ExitCode is
        return ExitSuccess();
    end;
end module body.
```

## Factorial

```austral
module Factorial is
    function main(): ExitCode;
end module.

module body Factorial is
    function fact(n: Nat64): Nat64 is
        if n = 0 then
            return 1;
        else
            return n * fact(n - 1);
        end if;
    end;

    function main(): ExitCode is
        let result: Nat64 := fact(5);
        result;
        return ExitSuccess();
    end;
end module body.
```

## Union and Case Analysis

```austral
module Example is
    function main(): ExitCode;
end module.

module body Example is
    union Number: Free is
        case Integer is
            value: Int32;
        case Float is
            value: Float64;
    end;

    function main(): ExitCode is
        let n: Number := Integer(value => 10);
        case n of
            when Integer(value: Int32) do
                skip;
            when Float(value: Float64) do
                skip;
        end case;
        return ExitSuccess();
    end;
end module body.
```

## Borrowing

```austral
module Example is
    function main(): ExitCode;
end module.

pragma Unsafe_Module;

module body Example is
    pragma Foreign_Import(External_Name => "putchar");
    function Put_Character(character: Nat8): Int32 is
    end;

    record Foo: Linear is
        bar: Nat8;
    end;

    function main(): ExitCode is
        let foo: Foo := Foo(bar => 'a');
        borrow fooref: &[Foo, rho] := &foo do
            let subref: &[Nat8, rho] := &(fooref->bar);
            let b: Nat8 := !subref;
            Put_Character(b);
        end borrow;
        let { bar: Nat8 } := foo;
        return ExitSuccess();
    end;
end module body.
```

## Interface and Body Pair

```austral
module Example is
    generic [T: Type]
    function identity(x: T): T;
end module.

module body Example is
    generic [T: Type]
    function identity(x: T): T is
        return x;
    end;
end module body.
```
